# create a tiny sample (5 minutes per vehicle) to avoid very large files
python - <<'PY'
from datetime import datetime, timedelta
import csv, random
VEHICLES = [f"VEH{100+i}" for i in range(10)]
start = datetime.utcnow() - timedelta(minutes=10)
with open('data/telemetry_sample.csv','w',newline='') as f:
    w = csv.writer(f)
    w.writerow(['timestamp','vehicle_id','speed_kmh','engine_temp_c','battery_v','odometer_km','fault_code'])
    for m in range(50):
        ts = (start + timedelta(minutes=m)).isoformat()+'Z'
        v = random.choice(VEHICLES)
        w.writerow([ts, v, int(random.gauss(60,15)), int(random.gauss(90,6)), round(random.gauss(12.6,0.1),2), random.randint(1000,60000), random.choice(['','P0010','P0300'])])
print("Wrote data/telemetry_sample.csv")
PY
